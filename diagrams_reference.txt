@startuml seq_login
skinparam backgroundColor #FAFAFA

actor       "User"           as U
participant "LoginPage"      as FE
participant "AuthController" as CTRL
participant "BetterAuth"     as BA
participant "BruteForce\nGuard (Redis)" as BF
database    "PostgreSQL"     as DB

U    -> FE   : Enter email + password
FE   -> CTRL : POST /api/auth/sign-in/email
CTRL -> BF   : checkRateLimit(ip, email)
alt Too many attempts
  BF --> CTRL : RATE_LIMIT_EXCEEDED
  CTRL --> FE : 429 Too Many Requests
  FE   --> U  : "Account locked. Try later."
else OK
  BF   --> CTRL: allowed
  CTRL -> BA   : signIn(email, password)
  BA   -> DB   : SELECT user WHERE email = ?
  alt Invalid credentials
    BA --> CTRL : AUTH_FAILED
    CTRL -> BF  : recordFailedAttempt(ip, email)
    CTRL --> FE : 401 Unauthorized
  else Valid — 2FA enabled
    BA --> CTRL : requires2FA
    CTRL --> FE : 200 { twoFactorRequired: true }
    FE   --> U  : Show TOTP input
    U    -> FE  : Enter TOTP code
    FE   -> CTRL: POST /api/auth/two-factor/verify
    CTRL -> BA  : verifyTOTP(code)
    BA   --> CTRL: OK
    CTRL --> FE : 200 + JWT token
    FE   --> U  : Redirect to dashboard
  else Valid — no 2FA
    BA --> CTRL : OK
    CTRL --> FE : 200 + JWT token
    FE   --> U  : Redirect to home
  end
end
@enduml

@startuml seq_checkout
skinparam backgroundColor #FAFAFA

actor       "Customer"         as C
participant "CheckoutPage"     as FE
participant "OrderController"  as CTRL
participant "CheckoutUseCase"  as UC
participant "CartRepo"         as CART
participant "ProductRepo"      as PROD
participant "OrderRepo"        as ORD
participant "TransactionMgr"   as TX
database    "PostgreSQL"       as DB

C   -> FE   : Fill shipping + click Place Order
FE  -> CTRL : POST /api/orders/checkout\n{shippingAddress, paymentMethod}
CTRL -> UC  : execute(userId, shippingAddress, paymentMethod)
UC  -> CART : findByUserId(userId)
CART -> DB  : SELECT cart items
CART --> UC : CartItems

UC  -> TX   : begin transaction
loop For each cart item
  TX -> PROD : findById(productId)
  PROD -> DB : SELECT product
  PROD --> TX : Product
  alt Insufficient stock
    TX --> UC : throw Error
    UC --> CTRL: failure(VALIDATION_ERROR)
    CTRL --> FE: 422 Insufficient stock
    FE --> C  : Show error
  else Stock OK
    TX -> PROD : update(product.stock -= qty)
    PROD -> DB : UPDATE products
  end
end

TX  -> ORD  : create(order)
ORD -> DB   : INSERT order + order_items
TX  -> CART : clearCart(userId)
CART -> DB  : DELETE cart items
TX  --> UC  : Order created
UC  --> CTRL: success(Order)
CTRL --> FE : 201 Created
FE  --> C   : Redirect to /orders
@enduml

@startuml seq_chatbot
skinparam backgroundColor #FAFAFA

actor       "Customer"         as C
participant "ChatWidget"       as FE
participant "ChatController"   as CTRL
participant "RagHttpsClient"   as RAG_C
participant "Injection Guard"  as GUARD
participant "RAG Service\n(Python)" as RAG
participant "Ollama LLM"       as LLM
database    "Vector Store"     as VS

C     -> FE    : Type question
FE    -> CTRL  : POST /api/chat {message}
CTRL  -> RAG_C : query(message)
RAG_C -> GUARD : checkLength(message)
alt Message too long
  GUARD --> RAG_C: REJECTED
  RAG_C --> CTRL : injection detected
  CTRL --> FE   : 400 Bad Request
  FE --> C      : "Message rejected"
else Length OK
  RAG_C -> GUARD: regexCheck(message)
  RAG_C -> GUARD: mlClassify(message)
  alt Injection detected
    GUARD --> RAG_C : INJECTION
    CTRL --> FE : 400 Bad Request
  else Clean
    RAG_C -> RAG  : POST /query {question}
    RAG   -> VS   : vectorSearch(embed(question))
    VS    --> RAG  : top-k documents
    RAG   -> LLM  : prompt(context + question)
    LLM   --> RAG : answer
    RAG   --> RAG_C: {answer}
    RAG_C --> CTRL : answer
    CTRL  --> FE  : 200 {answer}
    FE    --> C   : Display answer
  end
end
@enduml

@startuml class_domain
skinparam classAttributeIconSize 0
skinparam backgroundColor #FAFAFA
skinparam classFontName Arial

class User {
  +id: string
  +email: string
  +name: string
  +role: "customer"|"admin"|"super_admin"
  +emailVerified: boolean
  +twoFactorEnabled: boolean
  +isSuspended: boolean
  +createdAt: Date
}

class Product {
  +id: string
  +name: string
  +description: string
  +price: number
  +stock: number
  +isActive: boolean
  +hasVariants: boolean
  +categoryId: string
  +images: string[]
  +skinType: string[]
  +ingredients: string[]
  +createdAt: Date
}

class ProductVariant {
  +id: string
  +productId: string
  +name: string
  +stock: number
  +price: number
}

class Category {
  +id: string
  +name: string
  +slug: string
}

class Cart {
  +id: string
  +userId: string
  +items: CartItem[]
  +updatedAt: Date
}

class CartItem {
  +productId: string
  +variantId: string?
  +quantity: number
  +price: number
}

class Order {
  +id: string
  +userId: string
  +items: OrderItem[]
  +totalAmount: number
  +status: OrderStatus
  +shippingAddress: string
  +paymentMethod: string
  +stripePaymentIntentId: string?
  +createdAt: Date
}

class OrderItem {
  +productId: string
  +productName: string
  +quantity: number
  +priceAtPurchase: number
  +variantId: string?
}

class Review {
  +id: string
  +userId: string
  +productId: string
  +rating: 1..5
  +comment: string
  +createdAt: Date
}

class Wishlist {
  +id: string
  +userId: string
  +productIds: string[]
}

class BehaviourSignal {
  +id: string
  +userId: string
  +productId: string
  +signalType: "VIEW"|"WISHLIST"|"CART_ADD"|"PURCHASE"
  +createdAt: Date
}

enum OrderStatus {
  PENDING
  CONFIRMED
  SHIPPED
  DELIVERED
  CANCELLED
}

User       "1" -- "1"   Cart
User       "1" -- "*"   Order
User       "1" -- "*"   Review
User       "1" -- "1"   Wishlist
User       "1" -- "*"   BehaviourSignal
Cart       "1" *-- "*"  CartItem
Order      "1" *-- "*"  OrderItem
Order       --> OrderStatus
Product    "1" -- "*"   ProductVariant
Product    "1" -- "*"   Review
Product     --> Category
CartItem   --> Product
OrderItem  --> Product
BehaviourSignal --> Product
@enduml
