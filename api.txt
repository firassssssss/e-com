import axios from 'axios';

const API_BASE_URL = process.env.NEXT_PUBLIC_API_URL || 'http://localhost:5000/api';

export const api = axios.create({
  baseURL: API_BASE_URL,
  headers: {
    'Content-Type': 'application/json',
  },
});

// Analytics & Signals
export const sendSignal = (payload: { event: string; metadata: any }) => 
  api.post('/analytics/signal', payload);

export const getAdminAnalytics = () => 
  api.get('/admin/analytics/user-activity');

// Profile & Onboarding
export const updateProfile = (data: any) => 
  api.put('/profile/update', data);

export const getProfileLogs = (userId: string) => 
  api.get(`/admin/profile-logs/${userId}`);

// Products
export const getRecommendedProducts = () => 
  api.get('/products/recommendations');
